git init - ініціалізація репозиторія
git add . - додати всі зміни до індексу
git status - статус проєкту
git commit - створити комміт на основі проіндексованих змін
    -m "назва коміта" - скорочений варіант додавання заголовку
git commit --amend - змінити попередній коміт
    -m "назва коміта" - скорочений варіант зміни заголовку
git log - історія комітів
    --oneline - в одну строку
git reflog - історія змін комітів
git diff - подивитися незафіксовані зміни
git diff --staged - подивитися індексовані (зафіксовані зміни)

-----


git remote - команда для керування віддаленими репозиторіями
git remote -v - перевірка підв'язки віддаленого репозиторія
git push origin <гілка> - запушити зміни на відд.реп
git pull origin <гілка> - стягнути (зпулити) зміни з віддаленого репа
git clone <адреса віддаленого репозиторія>
